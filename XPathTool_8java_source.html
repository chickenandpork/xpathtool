145  /**
146  * Consume a list of matches, and for each node (entity or attribute) make a Strong.replaceFirst() or a String.replaceAll()
147  *
148  * This function is added as a convenience function to allow semi-contextual replacement of strings
149  * when converting other XML content -- this allows single atomic commands to rename components to
150  * avoid name-clashes and unintended replacements.
151  *
152  * @param nodelist list of nodes containing targets/matches to alter
153  * @param editMatch string to match for replacement
154  * @param editReplace string to replace for matches
155  * @param global whether to replace globally (like sed -e 's/X/y/g' -- the 'g' at the end) -- which is whether to use a Strong.replaceFirst() or a String.replaceAll()
156  * @return number of matches made
157  */
158  public static int subAttr(NodeList nodelist, String editMatch, String editReplace, boolean global)
159  {
160  // http://download.oracle.com/javase/1.5.0/docs/api/org/w3c/dom/Node.html
161  int res = 0;
162 
163  for(int index = 0; index < nodelist.getLength(); index ++)
164  {
165  res++;
166  Node aNode = nodelist.item(index);
167  switch (aNode.getNodeType())
168  {
169  case Node.ELEMENT_NODE:
170  System.out.println ("replacing: "+aNode.getNodeType());
171  if (global)
172  aNode.setTextContent(aNode.getTextContent().replaceAll(editMatch, editReplace));
173  else
174  aNode.setTextContent(aNode.getTextContent().replaceFirst(editMatch, editReplace));
175  break;
176 
177  case Node.ATTRIBUTE_NODE:
178  if (global)
179  aNode.setNodeValue(aNode.getTextContent().replaceAll(editMatch, editReplace));
180  else
181  aNode.setNodeValue(aNode.getTextContent().replaceFirst(editMatch, editReplace));
182  break;
183 
184  default:
185  System.out.println ("Not replacing: "+aNode.getNodeType());
186  res--;
187  }
188  }
189 
190  return res;
191  }
